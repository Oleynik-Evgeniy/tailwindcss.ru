import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.sepia-0': { '--tw-sepia': 'sepia(0)' },
        '.sepia': { '--tw-sepia': 'sepia(1)' },
      })
    }
  }
}

## Применение

Используйте утилиты `sepia` и `sepia-0` вместе с утилитой `filter`, чтобы определить, должен ли элемент отображаться как сепия или в полном цвете.

```html
<div class="filter sepia ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять фильтром сепия элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите сепия. Например, используйте `md:sepia-0`, чтобы применить утилиту `sepia-0` только при средних размерах экрана и выше.

```html
<div class="filter sepia **md:sepia-0** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).
