import plugin from 'tailwindcss/lib/plugins/ringOffsetColor'
import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin,
  transformProperties: ({ selector, properties }) => {
    properties['box-shadow'] = `0 0 0 var(--tw-ring-offset-width) var(--tw-ring-offset-color), var(--tw-ring-shadow)`
    return properties
  }
}


## Применение

Используйте утилиты `ring-offset-{color}`, чтобы изменить цвет смещения кольца. Обычно это делается, чтобы попытаться сопоставить смещение с родительским цветом фона, поскольку истинные смещения тени блока фактически невозможны в CSS.

```html violet
<template preview class="p-0">
  <div class="bg-violet-100 grid grid-cols-1 gap-6 justify-items-center p-8 ">
    <p class="font-semibold text-violet-600">bg-purple-100</p>
    <button tabindex="-1" class="focus:outline-none text-sm px-5 py-3 rounded-md font-semibold text-white bg-violet-500 ring ring-violet-600 ring-offset-4 ring-offset-violet-100">
      ring-offset-purple-100
    </button>
  </div>
</template>

<div class="... **bg-purple-100**">
  <button class="... ring ring-purple-600 **ring-offset-4** **ring-offset-purple-100**">
    ring-offset-purple-100
  </button>
</div>
```

## Адаптивность

Чтобы управлять цветом смещения кольца в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите цвета смещения кольца. Например, используйте `md:ring-offset-blue-500`, чтобы применить утилиту `ring-offset-blue-500` только при средних размерах экрана и выше.

```html
<button class="ring-2 ring-offset-2 ring-offset-blue-300 **md:ring-offset-blue-500**">
  <!-- ... -->
</button>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).

## Кастомизация

Вы можете настроить, какие утилиты цвета смещения кольца будут генерироваться, настроив вашу цветовую палитру с помощью ключа `colors` в разделе `theme` вашего файла `tailwind.config.js`:

```js
// tailwind.config.js
const colors = require('tailwindcss/colors')

module.exports = {
  theme: {
    colors: {
      gray: colors.blueGray,
      indigo: colors.indigo,
      red: colors.rose,
      yellow: colors.yellow,
    }
  }
}
```

Если вы хотите настроить только утилиты цвета смещения кольца, не затрагивая вашу глобальную цветовую палитру, используйте вместо этого ключ `ringOffsetColor`:

```js
// tailwind.config.js
const colors = require('tailwindcss/colors')

module.exports = {
  theme: {
    ringOffsetColor: {
      white: colors.white,
      pink: colors.fuchsia,
    }
  }
}
```

Дополнительные сведения о настройке темы по умолчанию смотрите в [документации по настройке темы](/docs/theme#customizing-the-default-theme).
