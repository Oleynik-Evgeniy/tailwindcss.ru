import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.backdrop-blur-0': { '--tw-backdrop-blur': 'blur(0)' },
        '.backdrop-blur-sm': { '--tw-backdrop-blur': 'blur(4px)' },
        '.backdrop-blur': { '--tw-backdrop-blur': 'blur(8px)' },
        '.backdrop-blur-md': { '--tw-backdrop-blur': 'blur(12px)' },
        '.backdrop-blur-lg': { '--tw-backdrop-blur': 'blur(16px)' },
        '.backdrop-blur-xl': { '--tw-backdrop-blur': 'blur(24px)' },
        '.backdrop-blur-2xl': { '--tw-backdrop-blur': 'blur(40px)' },
        '.backdrop-blur-3xl': { '--tw-backdrop-blur': 'blur(64px)' },
      })
    }
  }
}

## Применение

Используйте утилиты `backdrop-blur-{amount?}` вместе с утилитой `backdrop-filter`, чтобы размыть элемент.

```html
<div class="backdrop-filter backdrop-blur-lg ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять размытием фона элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите размытия фона. Например, используйте `md:backdrop-blur-lg`, чтобы применить утилиту `backdrop-blur-lg` только при средних размерах экрана и выше.

```html
<div class="backdrop-filter backdrop-blur-sm **md:backdrop-blur-lg** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).
