import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.grayscale-0': { '--tw-grayscale': 'grayscale(0)' },
        '.grayscale': { '--tw-grayscale': 'grayscale(1)' },
      })
    }
  }
}

## Применение

Используйте утилиты `grayscale` и `grayscale-0` вместе с утилитой `backdrop-filter`, чтобы определить, должен ли элемент отображаться в оттенках серого или в полноцветном.

```html
<div class="backdrop-filter grayscale ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять фильтром оттенки серого элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите оттенков серого. Например, используйте `md:grayscale-0`, чтобы применить утилиту `grayscale-0` только при средних размерах экрана и выше.

```html
<div class="filter grayscale **md:grayscale-0** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).

## Кастомизация

### Варианты

<Variants plugin="grayscale" name="grayscale" />

### Отключение

<Disabling plugin="grayscale" name="grayscale" />
