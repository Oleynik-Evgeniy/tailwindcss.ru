import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.backdrop-sepia-0': { '--tw-backdrop-sepia': 'sepia(0)' },
        '.backdrop-sepia': { '--tw-backdrop-sepia': 'sepia(1)' },
      })
    }
  }
}

## Применение

Используйте утилиты `backdrop-sepia` и `backdrop-sepia-0` вместе с утилитой `backdrop-filter`, чтобы определить, должен ли фон элемента отображаться как сепия или в полном цвете.

```html
<div class="backdrop-filter backdrop-sepia ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять фильтром фона сепия элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите фона сепия. Например, используйте `md:backdrop-sepia-0`, чтобы применить утилиту `backdrop-sepia-0` только при средних размерах экрана и выше.

```html
<div class="backdrop-filter backdrop-sepia **md:backdrop-sepia-0** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).

## Кастомизация

Вы можете настроить, какие утилиты `backdrop-sepia` будут генерироваться, используя ключ `backdropSepia` в разделе `theme` вашего файла `tailwind.config.js`.

```diff-js
  // tailwind.config.js
  module.exports = {
    theme: {
+     extend: {
+       backdropSepia: {
+         25: '.25',
+         75: '.75',
+       }
+     }
    }
  }
```

Дополнительные сведения о настройке темы по умолчанию смотрите в [документации по настройке темы](/docs/theme#customizing-the-default-theme).
