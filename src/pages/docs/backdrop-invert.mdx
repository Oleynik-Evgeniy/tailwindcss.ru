import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.backdrop-invert-0': { '--tw-backdrop-invert': 'invert(0)' },
        '.backdrop-invert': { '--tw-backdrop-invert': 'invert(1)' },
      })
    }
  }
}

## Применение

Используйте утилиты `backdrop-invert` и `backdrop-invert-0` вместе с утилитой `backdrop-filter`, чтобы определить, должен ли элемент отображаться с инвертированными цветами фона или в обычном режиме.

```html
<div class="backdrop-filter backdrop-invert ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять фильтром инвертирования фона элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите инвертирования фона. Например, используйте `md:backdrop-invert-0`, чтобы применить утилиту `backdrop-invert-0` только при средних размерах экрана и выше.

```html
<div class="backdrop-filter backdrop-invert **md:backdrop-invert-0** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).
