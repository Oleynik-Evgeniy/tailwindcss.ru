import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.saturate-0': { '--tw-saturate': 'saturate(0)' },
        '.saturate-50': { '--tw-saturate': 'saturate(.5)' },
        '.saturate-100': { '--tw-saturate': 'saturate(1)' },
        '.saturate-150': { '--tw-saturate': 'saturate(1.5)' },
        '.saturate-200': { '--tw-saturate': 'saturate(2)' },
      })
    }
  }
}

## Применение

Используйте утилиты `saturate-{amount}` вместе с утилитой `filter` для управления насыщенностью элемента.

```html
<div class="filter saturate-125 ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять насыщенностью элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите насыщенности. Например, используйте `md:saturate-150`, чтобы применить утилиту `saturate-150` только при средних размерах экрана и выше.

```html
<div class="filter saturate-110 **md:saturate-150** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).

## Кастомизация

Вы можете настроить, какие утилиты `saturate` будут генерироваться, используя ключ `saturate` в разделе `theme` вашего файла `tailwind.config.js`.

```diff-js
  // tailwind.config.js
  module.exports = {
    theme: {
+     extend: {
+       saturate: {
+         25: '.25',
+         75: '.75',
+       }
+     }
    }
  }
```

Дополнительные сведения о настройке темы по умолчанию смотрите в [документации по настройке темы](/docs/theme#customizing-the-default-theme).
