import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.backdrop-grayscale-0': { '--tw-backdrop-grayscale': 'grayscale(0)' },
        '.backdrop-grayscale': { '--tw-backdrop-grayscale': 'grayscale(1)' },
      })
    }
  }
}

## Применение

Используйте утилиты `backdrop-grayscale` и `backdrop-grayscale-0` вместе с утилитой `backdrop-filter`, чтобы определить, должен ли фон элемента отображаться в оттенках серого или в полноцветном.

```html
<div class="backdrop-filter backdrop-grayscale ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять фильтром фона оттенка серого элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите фона оттенка серого. Например, используйте `md:backdrop-grayscale-0`, чтобы применить утилиту `backdrop-grayscale-0` только при средних размерах экрана и выше.

```html
<div class="backdrop-filter backdrop-grayscale **md:backdrop-grayscale-0** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).

## Кастомизация

Вы можете настроить, какие утилиты `backdrop-grayscale` будут генерироваться, используя ключ `backdropGrayscale` в разделе `theme` вашего файла `tailwind.config.js`.

```diff-js
  // tailwind.config.js
  module.exports = {
    theme: {
+     extend: {
+       backdropGrayscale: {
+         50: '.5',
+       }
+     }
    }
  }
```

Дополнительные сведения о настройке темы по умолчанию смотрите в [документации по настройке темы](/docs/theme#customizing-the-default-theme).
