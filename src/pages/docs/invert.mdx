import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.invert-0': { '--tw-invert': 'invert(0)' },
        '.invert': { '--tw-invert': 'invert(1)' },
      })
    }
  }
}

## Применение

Используйте утилиты `invert` и `invert-0` вместе с утилитой `filter`, чтобы определить, должен ли элемент отображаться с инвертированными цветами или в обычном режиме.

```html
<div class="filter invert ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять фильтром инвертирования элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите инвертирования. Например, используйте `md:invert-0`, чтобы применить утилиту `invert-0` только при средних размерах экрана и выше.

```html
<div class="filter invert **md:invert-0** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).
