import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.backdrop-saturate-0': { '--tw-backdrop-saturate': 'saturate(0)' },
        '.backdrop-saturate-50': { '--tw-backdrop-saturate': 'saturate(.5)' },
        '.backdrop-saturate-100': { '--tw-backdrop-saturate': 'saturate(1)' },
        '.backdrop-saturate-150': { '--tw-backdrop-saturate': 'saturate(1.5)' },
        '.backdrop-saturate-200': { '--tw-backdrop-saturate': 'saturate(2)' },
      })
    }
  }
}

## Применение

Используйте утилиты `saturate-{amount}` вместе с утилитой `backdrop-filter` для управления насыщенностью фона элемента.

```html
<div class="backdrop-filter backdrop-saturate-125 ...">
  <!-- ... -->
</div>
```

## Адаптивность

Чтобы управлять насыщением фона элемента в определенной контрольной точке, добавьте префикс `{screen}:` к любой существующей утилите насыщения фона. Например, используйте `md:backdrop-saturate-150`, чтобы применить утилиту `backdrop-saturate-150` только при средних размерах экрана и выше.

```html
<div class="backdrop-filter backdrop-saturate-110 **md:backdrop-saturate-150** ...">
  <!-- ... -->
</div>
```

Для получения дополнительной информации о функциях адаптивного дизайна Tailwind ознакомьтесь с документацией [Адаптивного дизайна](/docs/responsive-design).

## Кастомизация

Вы можете настроить, какие утилиты `backdrop-saturate` будут генерироваться, используя ключ `backdropSaturate` в разделе `theme` вашего файла `tailwind.config.js`.

```diff-js
  // tailwind.config.js
  module.exports = {
    theme: {
+     extend: {
+       backdropSaturate: {
+         25: '.25',
+       }
+     }
    }
  }
```

Дополнительные сведения о настройке темы по умолчанию смотрите в [документации по настройке темы](/docs/theme#customizing-the-default-theme).
